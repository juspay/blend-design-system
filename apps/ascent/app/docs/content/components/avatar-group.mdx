## Usage

```tsx
import { AvatarGroup, AvatarSize, AvatarShape } from 'blend-v1'

function MyComponent() {
    const avatars = [
        { id: '1', alt: 'John Doe', src: 'https://example.com/john.jpg' },
        { id: '2', alt: 'Jane Smith', src: 'https://example.com/jane.jpg' },
        { id: '3', alt: 'Bob Johnson', fallback: 'BJ' },
        { id: '4', alt: 'Alice Brown', fallback: 'AB' },
        { id: '5', alt: 'Charlie Wilson', fallback: 'CW' },
    ]

    return (
        <AvatarGroup
            avatars={avatars}
            maxCount={3}
            size={AvatarSize.MD}
            shape={AvatarShape.CIRCULAR}
        />
    )
}
```

export type AvatarGroupTokensType = {
    gap: CSSObject['gap']
    container: {
        marginLeft: {
            [key in AvatarSize]: CSSObject['margin']
        }
    }
    avatar: {
        selected: {
            ringColor: CSSObject['borderColor']
            ringWidth: CSSObject['borderWidth']
            ringOffset: CSSObject['outlineOffset']
        }
        border: {
            width: CSSObject['borderWidth']
            color: CSSObject['borderColor']
        }
    }
    overflowCounter: {
        background: {
            default: CSSObject['backgroundColor']

            active: CSSObject['backgroundColor']
        }

            color: CSSObject['color']

            width: CSSObject['borderWidth']

        }

    }
}
