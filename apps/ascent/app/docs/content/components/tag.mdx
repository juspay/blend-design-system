## Usage

```tsx
import { Tag, TagVariant, TagColor, TagSize } from 'blend-v1'

function MyComponent() {
    return (
        <Tag
            text="Success"
            variant={TagVariant.SUBTLE}
            color={TagColor.SUCCESS}
            size={TagSize.MD}
            onClick={() => console.log('tag clicked')}
        />
    )
}
```
