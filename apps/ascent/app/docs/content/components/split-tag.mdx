## Usage

```tsx
import {
    SplitTag,
    TagColor,
    TagSize,
    TagShape,
    TagVariant,
} from '@juspay/blend-design-system'

function MyComponent() {
    return (
        <SplitTag
            primaryTag={{
                text: 'Status',
                variant: TagVariant.NO_FILL,
                color: TagColor.PRIMARY,
            }}
            secondaryTag={{
                text: 'Active',
                variant: TagVariant.ATTENTIVE,
                color: TagColor.SUCCESS,
            }}
            size={TagSize.MD}
            shape={TagShape.ROUNDED}
        />
    )
}
```
