## Usage

```tsx
import { Radio, RadioGroup, RadioSize } from 'blend-v1'
import { useState } from 'react'

function MyComponent() {
    const [selectedPlan, setSelectedPlan] = useState('basic')

    return (
        <RadioGroup
            name="subscription-plan"
            value={selectedPlan}
            onChange={setSelectedPlan}
            label="Choose your subscription plan"
        >
            <Radio
                value="basic"
                subtext="Perfect for individuals and small projects"
                slot={
                    <span className="text-green-600 text-sm font-medium">
                        Free
                    </span>
                }
            >
                Basic Plan
            </Radio>
            <Radio
                value="pro"
                subtext="Advanced features for growing teams"
                slot={
                    <span className="text-blue-600 text-sm font-medium">
                        $9/month
                    </span>
                }
            >
                Professional Plan
            </Radio>
            <Radio
                value="enterprise"
                subtext="Custom solutions for large organizations"
                slot={
                    <span className="text-purple-600 text-sm font-medium">
                        Contact Sales
                    </span>
                }
            >
                Enterprise Plan
            </Radio>
        </RadioGroup>
    )
}
```

## Component Tokens

You can style the Radio component using the following tokens:

```tsx
export type RadioTokensType = {
    gap: CSSObject['gap']
    group: {
        gap: CSSObject['gap']
    }
    indicator: {
        [key in RadioIndicatorState]: {
            backgroundColor: {
                [key in RadioState]: CSSObject['backgroundColor']
            }
            borderColor: {
                [key in RadioState]: CSSObject['borderColor']
            }
        }
    }
    activeIndicator: {
        active: {
            backgroundColor: {
                [key in Exclude<
                    RadioState,
                    'hover' | 'error'
                >]: CSSObject['backgroundColor']
            }
        }
    }
    content: {
        label: {
            gap: CSSObject['gap']
            color: {
                [key in RadioState]: CSSObject['color']
            }
            fontSize: {
                [key in RadioSize]: CSSObject['fontSize']
            }
            fontWeight: {
                [key in RadioSize]: CSSObject['fontWeight']
            }
        }
        sublabel: {
            color: {
                [key in RadioState]: CSSObject['color']
            }
            fontSize: {
                [key in RadioSize]: CSSObject['fontSize']
            }
            fontWeight: {
                [key in RadioSize]: CSSObject['fontWeight']
            }
        }
    }
    height: {
        [key in RadioSize]: CSSObject['height']
    }
    borderWidth: {
        [key in RadioIndicatorState]: {
            [key in RadioState]: number
        }
    }
    slot: {
        [key in RadioSize]: CSSObject['width']
    }
    required: {
        color: CSSObject['color']
    }
}
```
