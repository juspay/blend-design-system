## Usage

```tsx
import { SingleSelect, SelectMenuSize, SelectMenuVariant } from 'blend-v1'
import { useState } from 'react'
import { User, MapPin, Star } from 'lucide-react'

function MyComponent() {
    const [selectedUser, setSelectedUser] = useState('')

    const teamMembers = [
        {
            groupLabel: 'Developers',
            items: [
                {
                    label: 'Sarah Johnson',
                    value: 'sarah-johnson',
                    subLabel: 'Frontend Developer',
                    slot1: <User size={16} className="text-blue-500" />,
                    slot4: <Star size={14} className="text-yellow-500" />,
                },
                {
                    label: 'Mike Chen',
                    value: 'mike-chen',
                    subLabel: 'Backend Developer',
                    slot1: <User size={16} className="text-green-500" />,
                    slot4: <Star size={14} className="text-yellow-500" />,
                },
            ],
        },
        {
            groupLabel: 'Designers',
            items: [
                {
                    label: 'Alex Rivera',
                    value: 'alex-rivera',
                    subLabel: 'UI/UX Designer',
                    slot1: <User size={16} className="text-purple-500" />,
                    slot4: <Star size={14} className="text-yellow-500" />,
                },
            ],
        },
    ]

    return (
        <SingleSelect
            label="Assign team member"
            placeholder="Select a team member"
            items={teamMembers}
            selected={selectedUser}
            onSelect={setSelectedUser}
            enableSearch={true}
            searchPlaceholder="Search team members..."
            size={SelectMenuSize.MEDIUM}
            variant={SelectMenuVariant.CONTAINER}
            required={true}
            helpIconText="Choose a team member to assign this task"
        />
    )
}
```
