## Usage

```tsx
import { OTPInput } from '@juspay/blend-design-system'

function MyComponent() {
    const [otp, setOtp] = useState('')

    const handleOTPChange = (value: string) => {
        setOtp(value)
        if (value.length === 6) {
            // Handle complete OTP
            console.log('Complete OTP:', value)
        }
    }

    return (
        <OTPInput
            label="Verification Code"
            value={otp}
            onChange={handleOTPChange}
            length={6}
            autoFocus
            hintText="Enter the 6-digit code sent to your phone"
        />
    )
}
```

```tsx
            height: CSSObject['height']

                [key in OTPInputState]: CSSObject['border']

                [key in OTPInputState]: CSSObject['backgroundColor']

        }
    }
}

```
