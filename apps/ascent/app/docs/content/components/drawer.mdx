## Usage

```tsx
import {
    Drawer,
    DrawerTrigger,
    DrawerPortal,
    DrawerOverlay,
    DrawerContent,
    DrawerHeader,
    DrawerTitle,
    DrawerDescription,
    DrawerBody,
    DrawerFooter,
    DrawerClose,
    Button,
    ButtonType,
} from '@juspay/blend-design-system'
import { Menu } from 'lucide-react'
import { useState } from 'react'

function MyComponent() {
    const [isOpen, setIsOpen] = useState(false)

    return (
        <Drawer open={isOpen} onOpenChange={setIsOpen}>
            <DrawerTrigger>
                <Button
                    text="Open Menu"
                    buttonType={ButtonType.SECONDARY}
                    leadingIcon={<Menu size={16} />}
                />
            </DrawerTrigger>

            <DrawerPortal>
                <DrawerOverlay />
                <DrawerContent direction="left">
                    <DrawerHeader>
                        <DrawerTitle>Navigation Menu</DrawerTitle>
                        <DrawerDescription>
                            Choose from the options below to navigate
                        </DrawerDescription>
                    </DrawerHeader>

                    <DrawerBody>
                        <div className="space-y-4">
                            <button className="w-full text-left p-3 hover:bg-gray-100 rounded">
                                Dashboard
                            </button>
                            <button className="w-full text-left p-3 hover:bg-gray-100 rounded">
                                Projects
                            </button>
                            <button className="w-full text-left p-3 hover:bg-gray-100 rounded">
                                Settings
                            </button>
                        </div>
                    </DrawerBody>

                    <DrawerFooter>
                        <DrawerClose>
                            <Button
                                text="Close"
                                buttonType={ButtonType.SECONDARY}
                                fullWidth
                            />
                        </DrawerClose>
                    </DrawerFooter>
                </DrawerContent>
            </DrawerPortal>
        </Drawer>
    )
}
```

## Component Tokens

You can style the Drawer component using the following tokens:

```tsx
export type DrawerTokensType = {
    borderRadius: {
        topLeft: CSSObject['borderTopLeftRadius']
        topRight: CSSObject['borderTopRightRadius']
        bottomLeft: CSSObject['borderBottomLeftRadius']
        bottomRight: CSSObject['borderBottomRightRadius']
    }
    overlay: {
        backgroundColor: CSSObject['backgroundColor']
    }
    offset: {
        top: CSSObject['top']
        bottom: CSSObject['bottom']
        left: CSSObject['left']
        right: CSSObject['right']
    }
    content: {
        backgroundColor: CSSObject['backgroundColor']
        padding: {
            x: CSSObject['padding']
            y: CSSObject['padding']
        }
        handle: {
            backgroundColor: CSSObject['backgroundColor']
            borderRadius: CSSObject['borderRadius']
            width: CSSObject['width']
            height: CSSObject['height']
        }
    }
}

export type ResponsiveDrawerTokens = {
    [key in keyof BreakpointType]: DrawerTokensType
}
```
