## Usage

```tsx
import {
    Menu,
    MenuAlignment,
    MenuSide,
    MenuItemVariant,
    MenuItemActionType,
} from 'blend-v1'
import { Button, ButtonType } from 'blend-v1'
import { Settings, User, LogOut, ChevronDown } from 'lucide-react'

function MyComponent() {
    const menuItems = [
        {
            label: 'Account',
            items: [
                {
                    label: 'Profile',
                    subLabel: 'Manage your profile',
                    slot1: <User size={16} />,
                    onClick: () => console.log('Profile clicked'),
                },
                {
                    label: 'Settings',
                    subLabel: 'Configure preferences',
                    slot1: <Settings size={16} />,
                    onClick: () => console.log('Settings clicked'),
                },
            ],
        },
        {
            label: 'Actions',
            items: [
                {
                    label: 'Sign Out',
                    slot1: <LogOut size={16} />,
                    variant: MenuItemVariant.ACTION,
                    actionType: MenuItemActionType.DANGER,
                    onClick: () => console.log('Sign out clicked'),
                },
            ],
            showSeparator: true,
        },
    ]

    return (
        <Menu
            trigger={
                <Button
                    text="Account Menu"
                    buttonType={ButtonType.SECONDARY}
                    trailingIcon={<ChevronDown size={16} />}
                />
            }
            items={menuItems}
            alignment={MenuAlignment.END}
            side={MenuSide.BOTTOM}
            enableSearch={true}
            searchPlaceholder="Search menu..."
        />
    )
}
```
