## Usage

```tsx
import { KeyValuePair, KeyValuePairSize, KeyValuePairStateType } from 'blend-v1'

function MyComponent() {
    return (
        <KeyValuePair
            keyString="Name"
            value="John Doe"
            size={KeyValuePairSize.MEDIUM}
            keyValuePairState={KeyValuePairStateType.vertical}
        />
    )
}
```

## Component Tokens

You can style the KeyValuePair component using the following tokens:

```tsx
export type KeyValuePairTokensType = {
    gap: {
        vertical: CSSObject['gap']
        horizontal: CSSObject['gap']
    }

    key: {
        color: CSSObject['color']
        fontSize: CSSObject['fontSize']
        fontWeight: CSSObject['fontWeight']
        gap: CSSObject['gap']
    }

    value: {
        color: CSSObject['color']
        fontSize: {
            [key in KeyValuePairSize]: CSSObject['fontSize']
        }
        fontWeight: CSSObject['fontWeight']
        gap: CSSObject['gap']
    }
}

export type ResponsiveKeyValuePairTokens = {
    [key in keyof BreakpointType]: KeyValuePairTokensType
}
```
