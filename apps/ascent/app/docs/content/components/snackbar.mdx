## Usage

```tsx
import {
    Snackbar,
    addSnackbar,
    SnackbarVariant,
} from '@juspay/blend-design-system'

function MyComponent() {
    const showSnackbar = () => {
        addSnackbar({
            header: 'Success!',
            description: 'Your action was completed successfully.',
            variant: SnackbarVariant.SUCCESS,
            onClose: () => console.log('Snackbar closed'),
            actionButton: {
                label: 'Undo',
                onClick: () => console.log('Undo clicked'),
            },
        })
    }

    return (
        <div>
            <button onClick={showSnackbar}>Show Snackbar</button>
            <Snackbar />
        </div>
    )
}
```

## Component Tokens

You can style the snackbar component using the following tokens:

```tsx
export type SnackbarTokens = {
    backgroundColor: CSSObject['backgroundColor']
    borderRadius: CSSObject['borderRadius']
    padding: CSSObject['padding']
    maxWidth: CSSObject['maxWidth']
    boxShadow: CSSObject['boxShadow']
    gap: CSSObject['gap']
    infoIcon: {
        color: {
            [key in SnackbarVariant]: CSSObject['color']
        }
        height: CSSObject['height' | 'width']
    }
    content: {
        gap: CSSObject['gap']
        textContainer: {
            gap: CSSObject['gap']
            header: {
                color: {
                    [key in SnackbarVarious]: CSSObject['color']
                }
                fontSize: CSSObject['fontSize']
                fontWeight: CSSObject['fontWeight']
            }
            description: {
                color: {
                    [key in SnackbarVariant]: CSSObject['color']
                }
                fontSize: CSSObject['fontSize']
                fontWeight: CSSObject['fontWeight']
            }
        }
    }
    actions: {
        primaryAction: {
            color: {
                [key in SnackbarVariant]: CSSObject['color']
            }
            fontSize: CSSObject['fontSize']
            fontWeight: CSSObject['fontWeight']
        }
        closeButton: {
            height: CSSObject['height' | 'width']
            color: {
                [key in SnackbarVariant]: CSSObject['color']
            }
        }
    }
}
```
