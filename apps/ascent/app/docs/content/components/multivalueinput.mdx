## Usage

```tsx
import { MultiValueInput } from '@juspay/blend-design-system'

function MyComponent() {
    const [skills, setSkills] = useState(['JavaScript', 'React'])

    const handleTagAdd = (tag: string) => {
        setSkills((prev) => [...prev, tag])
    }

    const handleTagRemove = (tagToRemove: string) => {
        setSkills((prev) => prev.filter((tag) => tag !== tagToRemove))
    }

    return (
        <MultiValueInput
            label="Technical Skills"
            placeholder="Type a skill and press Enter"
            tags={skills}
            onTagAdd={handleTagAdd}
            onTagRemove={handleTagRemove}
            hintText="Add your technical skills"
        />
    )
}
```

## Component Tokens

You can style the MultiValueInput component using the following tokens:

```tsx
export type MultiValueInputTokensType = {
    gap: CSSObject['gap']
    label: {
        fontSize: CSSObject['fontSize']
        fontWeight: CSSObject['fontWeight']
        color: {
            [key in MultiValueInputState]: CSSObject['color']
        }
    }
    subLabel: {
        fontSize: CSSObject['fontSize']
        fontWeight: CSSObject['fontWeight']
        color: {
            [key in MultiValueInputState]: CSSObject['color']
        }
    }
    hintText: {
        fontSize: CSSObject['fontSize']
        fontWeight: CSSObject['fontWeight']
        color: {
            [key in MultiValueInputState]: CSSObject['color']
        }
    }
    helpIcon: {
        width: CSSObject['width']
        color: {
            [key in MultiValueInputState]: CSSObject['color']
        }
    }
    errorMessage: {
        fontSize: CSSObject['fontSize']
        fontWeight: CSSObject['fontWeight']
        color: CSSObject['color']
    }
    required: {
        color: CSSObject['color']
    }
    inputContainer: {
        fontSize: {
            [key in MultiValueInputSize]: CSSObject['fontSize']
        }
        fontWeight: {
            [key in MultiValueInputSize]: CSSObject['fontWeight']
        }
        gap: CSSObject['gap']
        borderRadius?: CSSObject['borderRadius']
        boxShadow: {
            [key in MultiValueInputState]: CSSObject['boxShadow']
        }
        padding: {
            x: {
                [key in MultiValueInputSize]: CSSObject['padding']
            }
            y: {
                [key in MultiValueInputSize]: CSSObject['padding']
            }
        }
        border: {
            [key in MultiValueInputState]: CSSObject['border']
        }
        color: {
            [key in MultiValueInputState]: CSSObject['color']
        }
        backgroundColor: {
            [key in MultiValueInputState]: CSSObject['backgroundColor']
        }
    }
}

export type ResponsiveMultiValueInputTokens = {
    [key in keyof BreakpointType]: MultiValueInputTokensType
}

export type MultiValueInputSize = 'sm' | 'md' | 'lg'
export type MultiValueInputState =
    | 'default'
    | 'hover'
    | 'focus'
    | 'error'
    | 'disabled'
```

The component integrates with the Tag component system for displaying individual values with consistent styling and interactions.
