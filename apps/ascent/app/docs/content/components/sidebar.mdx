## Usage

```tsx
import { Sidebar } from 'blend-v1'
import { Dashboard, Projects, Settings } from 'lucide-react'

function MyComponent() {
    const directoryData = [
        {
            id: '1',
            label: 'Dashboard',
            href: '/dashboard',
            icon: <Dashboard size={16} />,
        },
        {
            id: '2',
            label: 'Projects',
            href: '/projects',
            icon: <Projects size={16} />,
            children: [
                {
                    id: '2-1',
                    label: 'Active Projects',
                    href: '/projects/active',
                },
                {
                    id: '2-2',
                    label: 'Completed',
                    href: '/projects/completed',
                },
            ],
        },
        {
            id: '3',
            label: 'Settings',
            href: '/settings',
            icon: <Settings size={16} />,
        },
    ]

    return (
        <Sidebar
            data={directoryData}
            topbar={
                <div className="flex items-center justify-between w-full">
                    <h1 className="text-lg font-semibold">My Application</h1>
                    <button className="px-3 py-1 bg-gray-100 rounded">
                        Profile
                    </button>
                </div>
            }
            footer={
                <div className="flex items-center gap-2">
                    <div className="w-8 h-8 bg-gray-300 rounded-full" />
                    <span className="text-sm">John Doe</span>
                </div>
            }
        >
            <div className="p-6">
                <h2 className="text-xl font-bold mb-4">Main Content</h2>
                <p>Your main application content goes here.</p>
            </div>
        </Sidebar>
    )
}
```

## Component Tokens

You can style the Sidebar component using the following tokens:

```tsx
export type SidebarTokenType = {
    maxWidth: {
        withLeftPanel: CSSObject['maxWidth']
        withoutLeftPanel: CSSObject['maxWidth']
    }
    backgroundColor: CSSObject['backgroundColor']
    borderRight: CSSObject['border']
    leftPanel: {
        width: CSSObject['width']
        backgroundColor: CSSObject['backgroundColor']
        borderRight: CSSObject['border']
        padding: {
            x: CSSObject['padding']
            y: CSSObject['padding']
        }
        gap: CSSObject['gap']
        item: {
            width: CSSObject['width']
            borderRadius: CSSObject['borderTopLeftRadius']
            border: {
                [key in SidebarState]: CSSObject['border']
            }
            backgroundColor: {
                [key in SidebarState]: CSSObject['backgroundColor']
            }
        }
    }
    header: {
        backgroundColor: CSSObject['backgroundColor']
        padding: {
            x: CSSObject['padding']
            y: CSSObject['padding']
        }
        gap: CSSObject['gap']
        borderBottom: CSSObject['border']
        toggleButton: {
            backgroundColor: {
                [key in SidebarState]: CSSObject['backgroundColor']
            }
        }
    }
    directory: {
        gap: CSSObject['gap']
        paddingX: CSSObject['padding']
        paddingY: CSSObject['padding']
    }
    footer: {
        backgroundColor: CSSObject['backgroundColor']
        padding: {
            x: CSSObject['padding']
            y: CSSObject['padding']
        }
        borderTop: CSSObject['border']
    }
}

export type ResponsiveSidebarTokens = {
    [key in keyof BreakpointType]: SidebarTokenType
}

export type SidebarState = 'default' | 'hover' | 'active'
```
