## Usage

### Basic Usage

```tsx
import { Upload } from '@juspay/blend-design-system'

function MyComponent() {
    const handleDrop = (
        acceptedFiles: File[],
        fileRejections: FileRejection[]
    ) => {
        console.log('Accepted files:', acceptedFiles)
        console.log('Rejected files:', fileRejections)
    }

    return (
        <Upload
            label="Upload File"
            subLabel="Max size 5MB"
            accept={['.pdf', '.doc', '.docx']}
            maxSize={5 * 1024 * 1024}
            onDrop={handleDrop}
        />
    )
}
```

### Form Integration

Use `value` and `onChange` props for form integration:

**Single File:**

```tsx
const [file, setFile] = useState<File | null>(null)

<Upload
    label="Upload File"
    value={file}
    onChange={(value) => {
        setFile(value) // File | null
        // Extract metadata: value?.name, value?.type
    }}
/>
```

**Multiple Files:**

```tsx
const [files, setFiles] = useState<File[]>([])

<Upload
    label="Upload Files"
    multiple={true}
    value={files}
    onChange={(value) => {
        setFiles(value) // File[]
        // Extract metadata:
        const filenames = value.map(f => f.name)
        const filemimes = value.map(f => f.type)
    }}
/>
```
