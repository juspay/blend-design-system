## Usage

```tsx
import { Charts, ChartType, ChartLegendPosition, AxisType } from 'blend-v1'

function MyComponent() {
    const data = [
        {
            name: 'Jan',
            data: {
                sales: { primary: { label: 'Sales', val: 4000 } },
                revenue: { primary: { label: 'Revenue', val: 2400 } },
            },
        },
        {
            name: 'Feb',
            data: {
                sales: { primary: { label: 'Sales', val: 3000 } },
                revenue: { primary: { label: 'Revenue', val: 1398 } },
            },
        },
        {
            name: 'Mar',
            data: {
                sales: { primary: { label: 'Sales', val: 2000 } },
                revenue: { primary: { label: 'Revenue', val: 9800 } },
            },
        },
    ]

    return (
        <Charts
            chartType={ChartType.LINE}
            data={data}
            chartHeaderSlot={<h3>Monthly Performance Dashboard</h3>}
            legendPosition={ChartLegendPosition.TOP}
            xAxis={{
                label: 'Month',
                type: AxisType.DATE_TIME,
                showLabel: true,
            }}
            yAxis={{
                label: 'Value',
                type: AxisType.CURRENCY,
                showLabel: true,
            }}
        />
    )
}
```

### Sankey Charts (Flow Diagrams)

```tsx
import { Charts, ChartType } from 'blend-v1'

function SankeyChart() {
    const data = [
        {
            name: 'payment-flow',
            data: {
                sankeyData: {
                    primary: {
                        nodes: [
                            { id: 'Visa', name: 'Visa', color: '#1e3a8a' },
                            {
                                id: 'Received',
                                name: 'Received',
                                color: '#3B82F6',
                            },
                            {
                                id: 'Authorised',
                                name: 'Authorised',
                                color: '#10B981',
                            },
                        ],
                        links: [
                            {
                                source: 'Visa', // Can use string ID or numeric index
                                target: 'Received',
                                value: 165.0,
                                color: 'rgba(59, 130, 246, 0.4)',
                                hoverColor: 'rgba(59, 130, 246, 0.8)', // Optional hover effect
                            },
                            {
                                source: 'Received',
                                target: 'Authorised',
                                value: 220.0,
                                color: 'rgba(16, 185, 129, 0.4)',
                                hoverColor: 'rgba(16, 185, 129, 0.8)',
                            },
                        ],
                    },
                    aux: [],
                },
            },
        },
    ]

    return (
        <Charts
            chartType={ChartType.SANKEY}
            data={data}
            height={700}
            chartHeaderSlot={<h3>Payment Flow Diagram</h3>}
        />
    )
}
```

## Sankey Chart Features

Sankey charts have unique features for visualizing flows:

### Node and Link Customization

```tsx
// Nodes support:
{
    id: 'unique-id',        // String identifier (optional, can use numeric indices)
    name: 'Display Name',   // Visible label
    color: '#1e3a8a'       // Custom color for the node
}

// Links support:
{
    source: 'node-id',              // Source node (string ID or number)
    target: 'target-id',            // Target node (string ID or number)
    value: 100,                     // Flow magnitude
    color: 'rgba(59, 130, 246, 0.4)',      // Link color
    hoverColor: 'rgba(59, 130, 246, 0.8)'  // Color on hover (optional)
}
```

### Interactive Tooltips

Sankey charts display tooltips on hover:

- **Node Tooltip**: Shows node name and total value
- **Link Tooltip**: Displays source â†’ target flow with value

### Data Format Options

You can use either **numeric indices** or **string IDs** for links:

```tsx
// Option 1: Numeric indices (traditional)
links: [{ source: 0, target: 1, value: 100 }]

// Option 2: String IDs (more readable)
links: [{ source: 'Visa', target: 'Received', value: 165 }]
```

## Component Tokens

You can style the Charts component using the following tokens:

```tsx
export type ChartTokensType = {
    border: CSSObject['border']
    borderRadius: CSSObject['borderRadius']
    header: {
        padding: {
            x: CSSObject['padding']
            y: CSSObject['padding']
        }
        backgroundColor: CSSObject['backgroundColor']
        borderBottom: CSSObject['borderBottom']
        borderRadius: CSSObject['borderRadius']
        slots: {
            gap: CSSObject['gap']
        }
    }
    content: {
        legend: {
            gap: CSSObject['gap']
            dropdown: {
                maxHeight: CSSObject['maxHeight']
            }
            item: {
                gap: CSSObject['gap']
                color: {
                    [key in ChartState]: CSSObject['color']
                }
                fontSize: CSSObject['fontSize']
                fontWeight: CSSObject['fontWeight']
            }
        }
        padding: {
            top: CSSObject['padding']
            right: CSSObject['padding']
            bottom: CSSObject['padding']
            left: CSSObject['padding']
        }
        gap: CSSObject['gap']
        backgroundColor: CSSObject['backgroundColor']
    }
}

export type ResponsiveChartTokens = {
    [key in keyof BreakpointType]: ChartTokensType
}

export type ChartState = 'default' | 'hover' | 'active'
```
