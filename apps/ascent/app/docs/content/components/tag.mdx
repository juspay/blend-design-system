## Usage

```tsx
import { Tag, TagVariant, TagColor, TagSize } from 'blend-v1'

function MyComponent() {
    return (
        <Tag
            text="Success"
            variant={TagVariant.SUBTLE}
            color={TagColor.SUCCESS}
            size={TagSize.MD}
            onClick={() => console.log('tag clicked')}
        />
    )
}
```

## Component Tokens

You can style the Tag component using the following tokens:

```tsx
export type TagTokensType = {
    gap: CSSObject['gap']
    backgroundColor: {
        [key in TagVariant]: {
            [key in TagColor]: CSSObject['backgroundColor']
        }
    }
    borderRadius: {
        [key in TagSize]: {
            [key in TagShape]: CSSObject['borderRadius']
        }
    }
    padding: {
        [key in TagSize]: CSSObject['padding']
    }
    border: {
        [key in TagVariant]: {
            [key in TagColor]: CSSObject['border']
        }
    }
    text: {
        color: {
            [key in TagVariant]: {
                [key in TagColor]: CSSObject['color']
            }
        }
        fontSize: {
            [key in TagSize]: CSSObject['fontSize']
        }
        fontWeight: {
            [key in TagSize]: CSSObject['fontWeight']
        }
    }
}
```
