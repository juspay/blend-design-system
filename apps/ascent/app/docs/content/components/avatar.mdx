## Usage

```tsx
import { Avatar, AvatarSize, AvatarShape } from 'blend-v1'

function MyComponent() {
    return (
        <Avatar
            src="https://example.com/user-avatar.jpg"
            alt="John Doe"
            size={AvatarSize.MD}
            shape={AvatarShape.CIRCULAR}
            online={true}
        />
    )
}
```

## Component Tokens

You can style the avatar component using the following tokens:

```tsx
export type AvatarTokensType = {
    gap: CSSObject['gap']
    container: {
        size: {
            [key in AvatarSize]: {
                width: CSSObject['width']
                height: CSSObject['height']
            }
        }
        backgroundColor: {
            [key in 'withImage' | 'withoutImage']: {
                [key in 'default' | 'online']: CSSObject['backgroundColor']
            }
        }
        border: {
            [key in 'withImage' | 'withoutImage']: {
                [key in 'default' | 'online']: CSSObject['border']
            }
        }
        borderRadius: {
            [key in AvatarShape]: CSSObject['borderRadius']
        }
    }
    text: {
        color: {
            [key in 'default' | 'online']: CSSObject['color']
        }
        fontSize: {
            [key in AvatarSize]: CSSObject['fontSize']
        }
        fontWeight: {
            [key in AvatarSize]: CSSObject['fontWeight']
        }
    }
    indicator: {
        backgroundColor: {
            [key in 'default' | 'online']: CSSObject['backgroundColor']
        }
        border: {
            [key in 'default' | 'online']: {
                color: CSSObject['borderColor']
                width: CSSObject['borderWidth']
            }
        }
        size: {
            [key in AvatarSize]: {
                width: CSSObject['width']
                height: CSSObject['height']
            }
        }
        borderRadius: CSSObject['borderRadius']
        boxShadow: CSSObject['boxShadow']
    }
    slot: {
        spacing: CSSObject['margin']
        color: {
            [key in 'default' | 'online']: CSSObject['color']
        }
    }
}
```
